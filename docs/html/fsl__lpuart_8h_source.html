669 static inline uint8_t LPUART_ReadByte(LPUART_Type *base)
670 {
671 #if defined(FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT) && FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT
672  uint32_t ctrl = base->CTRL;
673  uint8_t result;
674  bool isSevenDataBits = (((ctrl & LPUART_CTRL_M7_MASK) != 0U) ||
675  (((ctrl & LPUART_CTRL_M7_MASK) == 0U) && ((ctrl & LPUART_CTRL_M_MASK) == 0U) &&
676  ((ctrl & LPUART_CTRL_PE_MASK) != 0U)));
677 
678  if (isSevenDataBits)
679  {
680  result = (uint8_t)(base->DATA & 0x7FU);
681  }
682  else
683  {
684  result = (uint8_t)base->DATA;
685  }
686 
687  return result;
688 #else
689  return (uint8_t)(base->DATA);
690 #endif
691 }
